fin_cols = ['Revenue', 'ProfitMargin', 'MarketCap', 'GrowthRate']
for col in fin_cols:
    Q1, Q3 = n[col].quantile([0.25, 0.75])
    IQR = Q3 - Q1
    lower, upper = Q1 - 1.5*IQR, Q3 + 1.5*IQR
    n = n[(n[col] >= lower) & (n[col] <= upper)]
